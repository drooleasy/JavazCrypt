1 /**
  2  * solve a 2nd-degree equation
  3  * @param  {[type]} a 2nd degre factor
  4  * @param  {[type]} b 1st degree factor
  5  * @param  {[type]} c constant factor
  6  * @return {Array} with the two -possibly equals- solutions or empty if no solution
  7  */
  8 function solveP2(a, b, c){ // ax2 + bx + c = 0, a!=0 !!!!
  9 	var det = b*b - 4*a*c;
 10 	if(det>=0){
 11 		return [
 12 			(-b + Math.sqrt(det))/(2*a),
 13 			(-b - Math.sqrt(det))/(2*a)
 14 		];
 15 	} else return [];
 16 }

 19 /**
 20  * compute the segment between an origin and and a point, from this point to a limit distance
 21  * @param  {number} ox    x-origin of ray
 22  * @param  {number} oy    y-origin of ray
 23  * @param  {number} tx    x of start/through point
 24  * @param  {number} ty    y of start/through point
 25  * @param  {number} limit max length of ray
 26  * @return {Segment}       the ray Segment or null if the point is too far
 27  */
 28 function castRay(ox, oy, tx, ty, limit){
 29 	var vx = tx-ox,
 30 		vy = ty-oy;
 31 
 32 
 33 	/*
 34 		tx,
 35 		ty,
 36 		tx + k * vx ,
 37 		ty + k * vy
 38 
 39 		dx = tx + k * vx - ox
 40 		dy = ty + k * vy - oy
 41 
 42 		dx*dx + dy*dy == limit*limit
 43 
 44 
 45 		(tx + k * vx - ox)*(tx + k * vx - ox)
 46 		(tx - ox + k * vx)*(tx - ox + k * vx)
 47 
 48 		(tx-ox)² + 2*(tx-ox)*(k*vx) + k² * vx²
 49 
 50 		k² * vx² + k² * vy²
 51 		2*(tx-ox)*(k*vx) + 2*(ty-oy)*(k*vy)
 52 		(tx-ox)² + (ty-oy)² - limit²
 53 
 54 		2*(tx-ox)*(k*vx) + 2*(ty-oy)*(k*vy)
 55 		2*tx*k*vx - 2*ox*k*vx + 2*ty*k*vy - 2*oy*k*vy
 56 		k*(2*tx*vx -2*ox*vx + 2*ty*vy - 2*oy*vy)
 57 	*/
 58 		var sol = solveP2(
 59 			(vx*vx + vy*vy),
 60 			(2*tx*vx -2*ox*vx + 2*ty*vy - 2*oy*vy),
 61 			((tx-ox)*(tx-ox) + (ty-oy)*(ty-oy) - limit*limit)
 62 		);
 63 
 64 		if(sol.length>0){
 65 			return new Segment(tx, ty, tx + sol[0] * vx, ty + sol[0] * vy);
 66 		}
 67 		return null
 68 
 69 }

 72 /**
 73  * compute cross product between two vector
 74  * @param  {object} a vector with x and y properties
 75  * @param  {object} b vector with x and y properties
 76  * @return {number}   the cross product
 77  */
 78 function crossProduct(a,b){
 79 	return a.x * b.y - a.y * b.x;
 80 }

107 /**
108  * compute intersction betwwen two ranges. a1<a2 && b1<b2
109  * @param {number} a1 first range start
110  * @param {number} a2 first range end
111  * @param {number} b1 second range start
112  * @param {number} b2 second range end
113  * @return {Array} the intersection start and end
114  */
115 function intersection(a1,a2,b1,b2){ // a1<a2 && b1<b2
116 
117 	if(b1>a2 || a1 >b2) return []
118 
119 	var mn = Math.min(a1,b1);
120 	var mx = Math.max(a2,b2);
121 
122 	if(a1>b1 && a2<b2) return [[b1,b2]];
123 	if(b1>a1 && b2<a2) return [[a1,a2]];
124 
125 	return [Math.max(mn, a1, b1), Math.min(mx, a2, b2)];
126 
127 }
128 /**
129  * compute union betwwen two ranges. a1<a2 && b1<b2
130  * @param {number} a1 first range start
131  * @param {number} a2 first range end
132  * @param {number} b1 second range start
133  * @param {number} b2 second range end
134  */
135 
136 function union(a1,a2,b1,b2){ // a1<a2 && b1<b2
137  * @return {Array} the union start and end or the thwo disjoint ranges
138 
139 	if(b1>a2 || a1 >b2) return [[a1,a2], [b1,b2]];
140 	var mn = Math.min(a1,b1);
141 	var mx = Math.max(a2,b2);
142 	return [mn,mx]
143 
144 }
