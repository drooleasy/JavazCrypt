 40 	
 41 	var angle_a = metrics_a.angle - bob_or_light.angle,
 42 		angle_b = metrics_b.angle - bob_or_light.angle;
 43 
 44 	var distance_a = metrics_a.distance,
 45 		distance_b = metrics_b.distance;
 46 
 47 	if(distance_a > bob_or_light.sightLength && distance_b > bob_or_light.sightLength){
 48 		return;
 49 	};
 50 
 51 	angle_a = clipAnglePositive(angle_a);
 52 	angle_b = clipAnglePositive(angle_b);
 53 	
 54 	
 55 	var mn = Math.min(angle_a, angle_b);
 56 	var mx = Math.max(angle_a, angle_b);
 57 	
 58 	
 59 	
 60 	var left,
 61 		right;
 62 	if(mn==angle_a){

 66 		left = this.b;
 67 		right = this.a
 68 	}

 72 	var tmp;
 73 	if(diff > Math.PI){
 74 		tmp = left;
 75 		left = right;
 76 		right = tmp;
 77 	}
