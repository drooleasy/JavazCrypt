 31 /**
 32  * give the number of full-turn in an angle
 33  * @param {number} angle the angle
 34  * @return {number} the (float) number of turns
 35  */
 36 function turns(angle){
 37 	return angle / (PIPI);
 38 }
 39 
 40 /**
 41  * clip an angle betwwen -180° and +180°
 42  * @param {number} angle the angle to clip
 43  * @return {number} the clipped angle
 44  */
 45 function clipAngle(angle){
 46 	var t = turns(angle);
 47 	t = (t<0 ? Math.ceil(t) : Math.floor(t));
 48 	angle -= t * PIPI;
 49 	if(angle < -Math.PI) angle = PIPI + angle;
 50 	if(angle > Math.PI) angle =  angle - PIPI;
 51 	return angle;
 52 	// angle > 0 == to the right
 53 	// angle < 0 == to the left
 54 }

 56 /**
 57  * clip an angle betwwen 0° and +360°
 58  * @param {number} angle the angle to clip
 59  * @return {number} the clipped angle
 60  */
 61 
 62 function clipAnglePositive(angle){
 63 	var turns = angle / (2*Math.PI);
 64 	turns = (turns<0 ? Math.ceil(turns) : Math.floor(turns));
 65 	angle -= turns * 2*Math.PI;
 66 	if(angle < 0) angle = PIPI + angle ;
 67 	return angle;
 68 }

 70 /**
 71  * clip an angle betwwen -360° and 0°
 72  * @param {number} angle the angle to clip
 73  * @return {number} the clipped angle
 74  */
 75 
 76 function clipAngleNegative(angle){
 77 	var turns = angle / PIPI;
 78 	turns = (turns<0 ? Math.ceil(turns) : Math.floor(turns));
 79 	angle -= turns * PIPI;
 80 	if(angle < 0) angle = PIPI + angle ;
 81 	return angle - PIPI;
 82 }

 85 /**
 86  * compute the angle between two points
 87  * @param  {number} from_x start point x-coordiante
 88  * @param  {number} from_y start point y-coordiante
 89  * @param  {number} to_x   other point x-coordiante
 90  * @param  {number} to_y   other point y-coordiante
 91  * @return {number}        the angle in radians, cliped to [-180°,180°]
 92  */
 93 function angleBetween(from_x,from_y, to_x,to_y){
 94 	var dx = to_x - from_x,
 95 		dy = to_y - from_y;
 96 	return clipAngle(Math.atan2(dy,dx));
 97 }

 99 /**
100  * compute the distance between two point
101  * @param  {number} from_x start point x-coordiante
102  * @param  {number} from_y start point y-coordiante
103  * @param  {number} to_x   other point x-coordiante
104  * @param  {number} to_y   other point y-coordiante
105  * @return {number}        the distance
106  */
107 function distanceBetween(from_x,from_y, to_x,to_y){
108 	var dx = to_x - from_x,
109 		dy = to_y - from_y;
110 	return Math.sqrt(dx*dx+dy*dy);
111 }

113 /**
114  * compute the distance and angle between two points in one swipe, for performance reason
115  * @param  {number} from_x start point x-coordiante
116  * @param  {number} from_y start point y-coordiante
117  * @param  {number} to_x   other point x-coordiante
118  * @param  {number} to_y   other point y-coordiante
119  * @return {object}        with properties distance and angle
120  */
121 
122 function distanceAndAngle(from_x,from_y, to_x,to_y){
123 	var dx = to_x - from_x,
124 		dy = to_y - from_y,
125 		distance = Math.sqrt(dx*dx+dy*dy),
126 		angle = clipAngle(Math.atan2(dy,dx));
127 	return {
128 		distance : distance,
129 		angle : angle
130 	};
131 }

133 /**
134  * compute the sin, cosine and quadrant of an angle
135  * @param {number} angle the angle
136  * @return {object} with properties cos, sin, x(1=left, 0=right), y(1=top, 0=bottom), isTop, isLeft, isRight, isBottom
137  */
138 function quadrant(angle){
139 
140 	var cos = Math.cos(angle),
141 		sin = Math.sin(angle),
142 		res = {
143 		cos: cos,
144 		sin: sin,
145 		x: (cos > 0 ? 0 : 1),
146 		y: (sin > 0 ? 1 : 0)
147 	};
148 
149 	res.isTop = (res.y == 0);
150 	res.isLeft = (res.x == 0);
151 	res.isDown = (res.y == 1);
152 	res.isRight = (res.x == 1);
153 
154 	return res;
155 }
