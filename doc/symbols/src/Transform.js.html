 39 Transform.mixin = function(that){
 40 	var t = new Transform();
 41 	that.x = t.x;
 42 	that.y = t.y;
 43 	that.angle = t.angle;
 44 	that.scale = t.scale;
 45 	that.transform = {
 46 		toLocal : function(/*ctx*/){
 47 			Transform.prototype.toLocal.apply(that, arguments);
 48 		},
 49 		toGlobal : function(/*ctx*/){
 50 			Transform.prototype.toGlobal.apply(that, arguments);
 51 		}
 52 	}
 53 	return that;
 54 }

 57 Transform.prototype.toLocal = function(ctx){
 58 	ctx.translate(this.x, this.y);
 59 	ctx.rotate(this.angle);
 60 	ctx.scale(this.scale, this.scale);
 61 	
 62 }
 63 Transform.prototype.toGlobal = function(ctx){
 64 	ctx.scale(1/this.scale, 1/this.scale);
 65 	ctx.rotate(-this.angle);	
 66 	ctx.translate(-this.x, -this.y);
