1 /**
  2  * Axialy Aligned Bounding Box model
  3  * @constructor
  4  * @param {number} x top-left x-coordiante
  5  * @param {number} y top-left y-coordiante
  6  * @param {number} w width
  7  * @param {number} h height
  8  */
  9 function AABB(x, y, w, h){
 10 	this.x = x;
 11 	this.y = y;
 12 	this.w = w;
 13 	this.h = h;
 14 };
 15 
 16 /**
 17  * Returns wheter an AABB intersects with another
 18  * @param {AABB} other the other AABB
 19  * @return {boolean} true is the two AABBs intersects, else false
 20  */
 21 AABB.prototype.intersects = function(other){
 22 	if(!other instanceof AABB) throw "argument is not an AABB";
 23 	var dontIntersects = (
 24 		this.x > other.x + other.w
 25 		|| this.x + this.w < other.x
 26 		|| this.y > other.y + other.h
 27 		|| this.y + this.h < other.y
 28 	);
 29 	return !dontIntersects;
 30 
 31 };
