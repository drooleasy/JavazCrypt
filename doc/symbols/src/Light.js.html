 63 Light.prototype.draw = function(paper, segments, bobs){		
 64 	
 65 
 66 	this.renderer.width = paper.width;
 67 	this.renderer.height = paper.height;
 68 
 69 	var ctx = this.renderer.getContext('2d');
 70 
 71 	this.x = this._x + randomDelta(this.positionVariation);
 72 	this.y = this._y + randomDelta(this.positionVariation);
 73 
 74 	var that = this;
 75 
 76 
 77 	this.shadow.clear();
 78 
 79 
 80 	var l_aabb = this.AABB();
 81 	// WORLD SHADOWS
 82 	if(segments){
 83 		for(i=0;i<segments.length;i++){
 84 
 85 			var segment = segments[i];
 86 			var s_aabb = segment.AABB();
 87 			if(!l_aabb.intersects(s_aabb)){ 
 88 				continue;	
 89 			}
 90 
 91 			segment.castShadow(this);
 92 		}
 93 	}
 94 	
 95 	// OTHERS SHADOWS	
 96 	if(bobs) for(i=0;i<bobs.length;i++){
 97 		var bob = bobs[i];
 98 		if(bob.light !== this){
 99 			var d = distanceBetween(this.x, this.y, bob.x, bob.y);
100 		
101 			if(
102 				d > bob.width
103 			){
104 				var sees_bob = bob && d < this.sightLength+bob.width;
105 				if(sees_bob){
106 					bob.castShadow(this);
107 				}
108 			}else{
109 				bob.castOverShadow(this);			
110 			}
111 		}
112 	}
113 	
114 	
115 
116 
117 	var oldCompositeOperation = ctx.globalCompositeOperation;
118 
119 
120 
121 	ctx.beginPath();
122 	ctx.fillStyle="#000";
123 	ctx.fillRect(0,0, ctx.width, ctx.height);
124 	ctx.closePath();
125 
126 
127 	// CLIPPING
128 	ctx.save();
129 	ctx.beginPath();
130 	ctx.arc(this.x, this.y, this.sightLength, 0, this.sightWidth);
131 	ctx.clip()
132 
133 	// DRAWS SHADOWS
134 	ctx.fillStyle="#000";
135 	ctx.globalCompositeOperation = "source-over";
136 	this.shadow.draw(this.renderer);
137 	ctx.fill();
138 	
139 	
140 	// draw lumens
141 	ctx.globalCompositeOperation = "darken";
142 	var grd=ctx.createRadialGradient(this.x, this.y, 0, this.x, this.y, this.sightLength);
143 	grd.addColorStop(0, this.lightColor);
144 	grd.addColorStop(this.startDecay + randomDelta(this.decayVariation), this.lightColor);
145 	grd.addColorStop(1, this.shadowColor);
146 	ctx.fillStyle = grd;			
147 	ctx.lineWidth = this.lineWidth;
148 	ctx.beginPath();
149 	ctx.fillRect(0,0,this.renderer.width,this.renderer.height);
150 
151 	ctx.globalCompositeOperation = "source-over";
152 	
153 	// light border hack
154 	ctx.strokeStyle="#000";
155 	ctx.lineWidth = "6"
156 	ctx.beginPath();
157 	ctx.arc(this.x, this.y, this.sightLength, 0, this.sightWidth);
158 	ctx.closePath();
159 	ctx.stroke();
160 
161 	// draws light bulb
162 	ctx.globalCompositeoperation="source-over";
163 	ctx.fillStyle="#999";
164 	ctx.beginPath();
165 	ctx.arc(this.x, this.y, 2+Math.random()*2-1, 0, Math.PI*2)
166 	ctx.fill();
167 	ctx.globalCompositeOperation = oldCompositeOperation;
168 	
169 	
170 	ctx.restore();
171 	
172 	var pctx = paper.getContext("2d");
173 	oldCompositeOperation = pctx.globalCompositeOperation;
174 	pctx.globalCompositeOperation = "lighten";
175 	pctx.drawImage(this.renderer,0,0);
176 	pctx.globalCompositeOperation = oldCompositeOperation;
177 	
178 
179 }

181 Light.prototype.AABB = function lightAABB(tolerance){
182 	var topLeft = {
183 		x : this.x - this.sightLength,
184 		y : this.y - this.sightLength
185 	}
186 	var w = this.sightLength*2;
187 	var h = this.sightLength*2;
188 	
189 	var w2 = w*tolerance;
190 	var h2 = h*tolerance;
191 	
192 	topLeft.x += (w-w2)/2;
193 	topLeft.y += (h-h2)/2;
194 
195 	return new AABB(
196 		topLeft.x,
197 		topLeft.y,
198 		w2,
199 		h2
200 	);
201 	
202 }
