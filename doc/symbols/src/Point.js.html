  2 /**
  3  * Model for 2D point and vectors
  4  * can take 0 parameter (for 0,0)
  5  * or an array of length 2 for x and y
  6  * or an object with x and y-properties
  7  * or two numbers for x and y
  8  * @constructor
  9  */
 10 var Point = ArgRouter.decorate(
 11 	{},
 12 
 13 	"", function(){
 14 		this.x = 0;
 15 		this.y = 0;
 16 	},
 17 	"arr", function(arr){
 18 		this.x = arr[0];
 19 		this.y = arr[1];
 20 	},
 21 	"obj.x.y", function(point){
 22 		this.x = point.x;
 23 		this.y = point.y;
 24 	},
 25 	"num, num", function(arr){
 26 		this.x = parseFloat(arguments[0] || 0);
 27 		this.y = parseFloat(arguments[1] || 0);
 28 	},
 29 	function Point(ctx){
 30 		//console.log("ctx");
 31 		//console.log(ctx);
 32 		ctx.__merge__(this);
 33 
 34 	}
 35 );

 38 /**
 39  * return an ArgRouter router for a Point
 40  * defined by either no arg, an x-/y-object, two numbers or an array
 41  * @param {string} key the key to witch assign the point in the ArgRouter context
 42  * @return {object} a Point route
 43  */
 44 Point.route = function(key){
 45 	return {
 46 		"":function(){
 47 			this[key] = new Point(0,0);
 48 		},
 49 		"obj.x.y": function(pseudopoint){
 50 			this[key] = pseudopoint.clone();
 51 		},
 52 		"num, num": function(x,y){
 53 			this[key] = new Point(parseFloat(x),parseFloat(y));
 54 		},
 55 		"arr":function(arr){
 56 			this[key] = new Point(parseFloat(arr[0]), parseFloat(arr[1]));
 57 		}
 58 	}
 59 }

125 /**
126  * compute distance and angle betwwen the point and another
127  * @method
128  * @return {object}         with distanceand angle properties
129  */
130 Point.prototype.distanceAndAngle =  ArgRouter.decorate(
131 	{
132 		from : new Point(0,0)
133 	},
134 	ArgRouter.combine(Point.route("from")),
135 	function distanceAndAngle(ctx){
136 		var dx = ctx.from.x - this.x,
137 			dy = ctx.from.y - this.y,
138 			distance = Math.sqrt(dx*dx+dy*dy),
139 			angle = Point.clipAngle(Math.atan2(dy,dx));
140 		return {
141 			distance : distance,
142 			angle : angle
143 		};
144 	}
145 );

156 		var dx = ctx.from.x - this.x,

159

203 Point.prototype.scale = ArgRouter.decorate(
204 	{
205 		amount : 1,
206 		from : new Point(0,0)
207 	},
208 	ArgRouter.combine(
209 		{
210 			"":function(){},
211 			"num": function(_amount){
212 				this.amount = _amount;
213 			}
214 		},
215 		Point.route("from")
216 	),
217 	function  scale(ctx){
218 
219 		var metrics = this.distanceAndAngle(ctx.from.x, ctx.from.y);
220 
221 		this.x = ctx.from.x + Math.cos(metrics.angle) * metrics.distance * ctx.amount;
222 		this.y = ctx.from.y + Math.cos(metrics.angle) * metrics.distance * ctx.amount;
223 		return this;
224 	}

228 Point.prototype.rotate = ArgRouter.decorate(
229 	{
230 		angle : null,
231 		distance : null,
232 		from : new Point(0,0)
233 	},
234 	ArgRouter.combine(
235 		{
236 			"": function(){
237 				this.angle = 0;
238 				this.distance = this.__this__.distance();
239 			},
240 			"num": function(_angle){
241 				this.angle = _angle;
242 				this.distance = this.__this__.distance();
243 			},
244 			"num, num": function(_angle, _distance){
245 				this.angle = _angle;
246 				this.distance = _distance;
247 			}
248 		},
249 		Point.route("from")
250 	),
251 	function rotate(ctx){
252 		this.x = ctx.from.x + Math.cos(ctx.angle) * ctx.distance;
253 		this.y = ctx.from.x + Math.sin(ctx.angle) * ctx.distance;
254 		return this;
255 	}

276 /**
277  * return an angle cliped betwwen -PI and +PI
278  */
279 Point.clipAngle = function(angle){
280 	var pipi =  2*Math.PI;
281 	var t = angle / pipi;
282 	t = (t<0 ? Math.ceil(t) : Math.floor(t));
283 	angle -= t * pipi;
284 	if(angle < -Math.PI) angle = pipi + angle;
285 	if(angle > Math.PI) angle =  angle - pipi;
286 	return angle;
287 }
