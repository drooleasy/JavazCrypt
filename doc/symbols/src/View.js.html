 22 View.prototype.draw = function draw(){
 23 
 24 
 25 	// CLEARING
 26 	var ctx = paper.getContext("2d");
 27 
 28 
 29 	ctx.fillStyle = "#000";
 30 	ctx.fillRect(0, 0, paper_width, paper_height);
 31 	
 32 	world.player.shadow.clear();	
 33 	world.player.tints.clear();	
 34 	
 35 	ctx.save();
 36 
 37 	// TRANSFORMS
 38 	if(this.relative){
 39 	
 40 		ctx.translate(paper.width/2,paper.height/2);	
 41 		if(this.relative_angle) ctx.rotate(-this.world.player.angle-Math.PI/2);
 42 		ctx.translate(-this.world.player.x,-this.world.player.y);	
 43 	}	
 44 	
 45 	// GET LAST RENDERING
 46 	if(this.lastValidBuffer){ 
 47 		var oldCompo = ctx.globalCompositeOperation;
 48 		paper.getContext("2d").drawImage(this.lastValidBuffer,0,0);	
 49 		ctx.globalCompositeOperation = oldCompo;
 50 		
 51 	}
 52 
 53 	// DRAWS FOV	
 54 	if(this.draw_sight) this.world.player.drawSight(this.paper, this.world);
 55 	
 56 	// DRAWS OTHERS IF PERCEIVED
 57 	for(var i=0; i< this.world.bobs.length; i++){
 58 		var other = this.world.bobs[i];
 59 		if(other !== this.world.player){
 60 			var sees_bob = this.world.player.sees(other, this.world.allSegments());
 61 			if(!this.lights_on || !this.draw_sight || sees_bob || this.world.player.feels(other)){
 62 				other.draw(this.paper);
 63 			}
 64 		}
 65 	}
 66 	// PLAYERS REACTION
 67 	this.world.player.speak(this.paper, this.world.player, this.relative_angle);
 68 	
 69 	// OTHER ATTENTION
 70 	for(var i=0; i< this.world.bobs.length; i++){
 71 		var other = this.world.bobs[i];
 72 		if(other !== this.world.player){
 73 	
 74 			if(this.world.player.saying || other.feels(this.world.player)){
 75 				var a = clipAngle(
 76 					clipAnglePositive(angleBetween(other.x, other.y, this.world.player.x, this.world.player.y)) - clipAnglePositive(other.angle)
 77 				);
 78 				if(Math.abs(a) > deg2rad(2)) other.angle += deg2rad(2) * Math.abs(a)/a;
 79 			}
 80 
 81 			// OTHERS REACTION
 82 			var sees_player = other.sees(this.world.player, this.world.allSegments());
 83 			if(sees_player){
 84 				if(!old_sees_player) other.say(paper, "Hello Bob !");
 85 			}else{
 86 				old_sees_player = false;
 87 			}
 88 			other.speak(this.paper, this.world.player, this.relative_angle);
 89 		}
 90 	}
 91 	
 92 	// DRAWS PLAYER
 93 	this.world.player.draw(this.paper);
 94 
 95 	// REVERSE TRANSFORMS
 96 	//if(relative){
 97 	//	ctx.translate(player.x,player.y);
 98 	//	if(relative_angle) ctx.rotate(player.angle-Math.PI/2);
 99 	//	ctx.translate(-paper.width/2,-paper.height/2);	
100 			
101 	//}	
102 	ctx.restore();
103 	
104 	// REQUEST NEXT FRAME
105 	var that = this;
106 	window.requestAnimationFrame(function(){that.draw()});
107 }

110 View.prototype.renderScene = function renderScene(paper, world){
111 
112 	var all_segments = world.allSegments();
113 
114 	var slowBuffer = document.createElement("canvas");
115 	slowBuffer.width = paper.width;
116 	slowBuffer.height = paper.height;
117 
118 	var worldRenderer = document.createElement("canvas");
119 	worldRenderer.width = paper.width;
120 	worldRenderer.height = paper.height;
121 
122 
123 	var slowTempoDelay = 1000/30;
124 
125 	var that = this;
126 		
127 
128 
129 	slowTempo = function slowTempo(){
130 		var ctx = slowBuffer.getContext("2d");
131 		
132 		ctx.fillStyle="#000";
133 		ctx.fillRect(0,0,slowBuffer.width,slowBuffer.height);
134 
135 		
136 		var renderedLights = [],
137 			i=0,
138 			l=world.lights.length,
139 			n=l+1;
140 		
141 		
142 		function drawScene(ctx, nofill){
143 			// DRAWS SCENE
144 			ctx.fillStyle = "#333";
145 			ctx.strokeStyle = "#cfc";
146 			if(nofill) ctx.fillStyle = "rgba(0,0,0,0)";
147 			ctx.lineWidth = 4;
148 			for(var i=0;i<that.world.paths.length;i++){
149 				ctx.beginPath();
150 				that.world.paths[i].draw(worldRenderer, false);
151 				if(!nofill) ctx.fill();
152 				ctx.stroke();
153 			}
154 			ctx.fillStyle = "#000";
155 			for(var i=0;i<that.world.boulders.length;i++){
156 
157 
158 				ctx.beginPath();
159 				that.world.boulders[i].draw(worldRenderer, true);
160 				if(!nofill) ctx.fill();
161 				ctx.stroke();
162 			}
163 			for(var i=0; i< that.world.segments.length;i++){
164 				ctx.strokeStyle = "#cfc";
165 				if(world.segments[i] instanceof Glass) ctx.strokeStyle="rgba(255,255,255, 0.3)" 
166 				if(world.segments[i] instanceof Door) ctx.strokeStyle="#cc6" 
167 				ctx.beginPath();
168 				that.world.segments[i].draw(worldRenderer);
169 				ctx.stroke();
170 			}
171 		}
172 		
173 		function conclude(){
174 			
175 			var ctx = slowBuffer.getContext("2d");
176 			var wctx = worldRenderer.getContext("2d");
177 
178 
179 			if(that.lights_on){
180 
181 				wctx.globalCompositeOperation = "luminosity";
182 				wctx.drawImage(slowBuffer,0,0);
183 
184 				// re draw wall 
185 				wctx.globalCompositeOperation = "source-over";
186 				drawScene(wctx, true);
187 
188 			}
189 			that.lastValidBuffer.getContext("2d").putImageData(wctx.getImageData(0,0,worldRenderer.width, worldRenderer.height), 0, 0);
190 			setTimeout(slowTempo, slowTempoDelay);
191 		}
192 		
193 		
194 		setTimeout(function(){
195 			var ctx = worldRenderer.getContext("2d");
196 			ctx.globalCompositeOperation = "source-over";
197 			
198 			ctx.fillStyle="#393";
199 			ctx.fillRect(0,0,worldRenderer.width,worldRenderer.height);
200 			drawScene(ctx);
201 
202 			n--;
203 			if(n==0){
204 				conclude();
205 			}
206 		},0);
207 		
208 		for(i=0;i<l;i++){
209 			setTimeout(
210 				(function(a_light){
211 					return function(){
212 						a_light.draw(slowBuffer, world.allSegments(), world.bobs);
213 						n--;
214 						if(n==0){
215 							conclude();
216 						}
217 						
218 					};
219 				})(world.lights[i]),
220 			0)
221 		}
222 	}
223 
224 
225 	setTimeout(slowTempo, slowTempoDelay);
226 }
