  4 Bubble.style = {
  5 	"margin": 5,
  6 	"padding": 3,
  7 	"line-height": 11,
  8 	"fill": "#FFF",
  9 	"stroke": "#FFF",
 10 	"stroke-width": 1,
 11 	"text-fill": "#000",
 12 	minAnchorLength : 5
 13 }
 14 
 15 Bubble.draw = function (paper, msg, angle, offset) {
 16 	var anchor_start_x = Math.cos(angle) * offset,
 17 		anchor_start_y = Math.sin(angle) * offset;
 18 
 19 	var lines = msg.split("\n");
 20 	var longestLine = "";
 21 	for(var i=0;i<lines.length;i++){
 22 		if(lines[i].length > longestLine.length) longestLine = lines[i];
 23 	} 
 24 	
 25 	var margin = Bubble.style.margin;		
 26 	var padding = Bubble.style.padding;		
 27 	
 28 	var content_height = 2*Bubble.style.padding + Bubble.style["line-height"] * lines.length;
 29 	
 30 	var metrics = ctx.measureText(longestLine);
 31 
 32 	var bubble_width = metrics.width + 2*margin + 2*padding;
 33 	var bubble_height = content_height + 2*margin + 2*padding;
 34 
 35 	var min_anchor = Bubble.style.minAnchorLength,
 36 		maxLength = Math.max(bubble_width, bubble_height),
 37 		bubble_center_x = anchor_start_x + Math.cos(angle)*(min_anchor + maxLength/2),
 38 		bubble_center_y = anchor_start_y + Math.sin(angle)*(min_anchor + maxLength/2),
 39 		bubble_x = bubble_center_x - bubble_width/2,
 40 		bubble_y = bubble_center_y - bubble_height/2;
 41 
 42 	var text_x = bubble_x + margin + padding;
 43 	var text_y = bubble_y + margin + padding + Bubble.style["line-height"];
 44 
 45 
 46 
 47 	ctx.fillStyle   = Bubble.style["fill"];
 48 	ctx.strokeStyle = Bubble.style["stroke"];
 49 	ctx.lineWidth   = Bubble.style["stroke-width"];
 50 	ctx.fontColor   = Bubble.style["font-color"];
 51 	
 52 	var anchor = new Segment(
 53 		bubble_center_x,
 54 		bubble_center_y,
 55 		anchor_start_x,
 56 		anchor_start_y
 57 	);
 58 	
 59 	anchor.draw(paper);
 60 
 61 	drawRoundedRect(ctx, bubble_x, bubble_y, bubble_width, bubble_height);
 62 
 63 	ctx.fill();
 64 	ctx.stroke();
 65 
 66 	
 67 	ctx.fillStyle = Bubble.style["text-fill"]; 
 68 	for(var i=0; i<lines.length;i++){
 69 		ctx.fillText(
 70 			lines[i],
 71 			text_x, 
 72 			text_y + i * Bubble.style["line-height"]
 73 		);
 74 	}
 75 }
