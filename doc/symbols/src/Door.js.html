 64 Door.prototype.closestPointFrom = function(x,y){
 65 	
 66 	function closest_point_on_seg(seg, circ_pos){
 67 		var seg_v = {x:(seg.b.x - seg.a.x), y:(seg.b.y - seg.a.y)},
 68 			pt_v = {x:(circ_pos.x - seg.a.x), y:(circ_pos.y - seg.a.y)},
 69 			seg_v_len = Math.sqrt(seg_v.x*seg_v.x + seg_v.y*seg_v.y);
 70 		if(seg_v_len < 0){
 71 			throw new Error("Invalid segment length");
 72 		}
 73 		var seg_v_unit = {x:(seg_v.x / seg_v_len), y:(seg_v.y / seg_v_len)},
 74 			proj = pt_v.x * seg_v_unit.x + pt_v.y * seg_v_unit.y; // dot product
 75 		if(proj <= 0){
 76 			return {x: seg.a.x, y:seg.a.y};
 77 		}
 78 		if(proj >= seg_v_len){
 79 			return {x: seg.b.x, y:seg.b.y};
 80 		}
 81 		var proj_v = {x:seg_v_unit.x * proj, y:seg_v_unit.y * proj},
 82 			closest = {x:proj_v.x + seg.a.x, y:proj_v.y + seg.a.y};
 83 		return closest
 84 	}
 85 
 86 	return closest_point_on_seg(this, {x:x,y:y});
 87 
 88 }

106 Door.prototype.subSegments = function(){
107 	var metrics = distanceAndAngle(this.a.x, this.a.y, this.b.x, this.b.y),
108 		cos = Math.cos(metrics.angle),
109 		sin = Math.sin(metrics.angle),
110 		cos2 = Math.cos(Math.PI + metrics.angle),
111 		sin2 = Math.sin(Math.PI + metrics.angle),
112 		d= metrics.distance;
113 		
114 	var left = d*(1-this.openess)/2 - this.yawness;
115 	var right = d*(1-this.openess)/2 -this.yawness;
116 	
117 	right *= 1 - this.center;
118 	left  *= 1 + this.center;
119 	var middle = {
120 		x:this.a.x   + cos * left,
121 		y: this.a.y  + sin * left,
122 		x2:this.b.x  + cos2 * right,
123 		y2: this.b.y + sin2 * right
124 	}
125 
126 	return [
127 		new Segment(this.a.x, this.a.y, middle.x, middle.y),
128 		new Segment(middle.x2, middle.y2, this.b.x, this.b.y)
129 	]
130 }
