  2 function Path (/*x,y,...*/){
  3 	var i = 0,
  4 		l = arguments.length;
  5 	this.points = [];
  6 	
  7 	for(;i<l;i+=2){
  8 		this.points.push({
  9 			x:(arguments[i]),
 10 			y:(i+1<l ? arguments[i+1] : 0)
 11 		});
 12 	}
 13 	
 14 	this.segments = [];
 15 	l = this.points.length-1; 
 16 	for(i=0;i<l;i++){
 17 		if(false && i>0){
 18 			this.segments.push(new Segment(  // dummy welding to compensate line offsets
 19 				this.points[i-1].x, this.points[i-1].y,
 20 				this.points[i].x, this.points[i].y
 21 			));
 22 		}
 23 	
 24 		this.segments.push(new Segment(
 25 			this.points[i].x, this.points[i].y,
 26 			this.points[i+1].x, this.points[i+1].y
 27 		));
 28 	}
 29 	this.closed = false;
 30 }
